# Запрашиваем у пользователя начало и конец диапазона
start = int(input("Введите начало диапазона: "))
end = int(input("Введите конец диапазона: "))

# Проверяем, что начало меньше конца
if start > end:
    print("Ошибка: начало диапазона должно быть меньше конца")
else:
    # Находим числа Фибоначчи в диапазоне
    fibonacci_numbers = []
    
    # Первые два числа Фибоначчи
    a, b = 0, 1
    
    # Генерируем числа Фибоначчи, пока они не превысят конец диапазона
    while a <= end:
        if a >= start:
            fibonacci_numbers.append(a)
        a, b = b, a + b
    
    # Выводим результат
    if fibonacci_numbers:
        # Преобразуем числа в строки и объединяем через пробел
        result = " ".join(str(num) for num in fibonacci_numbers)
        print(result)
    else:
        print("В заданном диапазоне нет чисел Фибоначчи")
